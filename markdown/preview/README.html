What It Does

Parses a properly formated URL string to extract its scheme, authority (username & password), subdomains, domain, tld, hostname, port, path, query (parameters), and fragment.

What It Does Not Do

Does not parse URLs in the form of an IP address.

Add To Project

<script src="path/to/lib.js"></script>

Access Library

var parseURL = window.app.libs.parseURL;

Usage

For an exhaustive list of URLs see app.js. Run index.html and open up the console.

        "domain": true

    "punct": {
        "left": "",
        "right": ""
    },
    "mdomain": "youtube.com",
    "scheme": "https://",
    "protocol": "https",
    "userinfo": null,
    "username": null,
    "password": null,
    "subdomains": "www",
    "domain": "youtube",
    "tld": "com",
    "port": null,
    "pathname": "/watch",
    "query": "v=Gj2nOk8af-o",
    "parameters": {
        "v": "Gj2nOk8af-o"
    },
    "fragment": null,
    "hash": null,
    "authority": "www.youtube.com",
    "url": "https://www.youtube.com/watch?v=Gj2nOk8af-o",

    "directory": "/watch",
    "filename": null,
    "hostname": "www.youtube.com",
    "host": "www.youtube.com",
    "origin": "https://www.youtube.com",

        "subtype": null,
